\section{Introduction}

\begin{frame}{Signatures?}
    \begin{itemize}
        \item What do we mean when we talk about “signatures”? \pause
        \item A factor or set of factors that define or identify some (biological) phenomenon
    \end{itemize}
\end{frame}

\begin{frame}{Signatures vs Biomarkers}
    \begin{columns}[t]
        \begin{column}{0.5\textwidth}
            \begin{block}{Biomarkers}
                \begin{itemize}
                    \item Biomarkers are objective and measurable
                    \item Usually corresponding to single factors (a gene, molecule etc)
                    \item Predictive or diagnostic power
                    \item Established causal link, well characterised values under normal conditions
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{Signatures}
                \begin{itemize}
                    \item Signatures can represent combinations of factors
                    \item Not always directly measurable
                    \item Not always informed by causal links
                    \item Not always selected for predictive/diagnostic ability
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\end{frame}

\begin{frame}{How will we approach this?}
    \begin{itemize}
        \item ``A factor or set of factors that define or identify some (biological) phenomenon''
        \item Factors $\implies$ variables
        \item Signature: pattern of/relationship between variables
        \item Signature discovery: process for identifying those patterns
    \end{itemize}
\end{frame}

\begin{frame}{Two Distinct Scenarios}
    \begin{columns}[t]
        \begin{column}{0.5\textwidth}
            \begin{block}{Supervised}
                The phenomena we are looking to identify signatures for are known, and some samples are "labelled"
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{Unsupervised}
                The phenomena we are looking to identify signatures for are \textbf{not} known, \textbf{and/or} and no samples are "labelled"
            \end{block}
        \end{column}

    \end{columns}
\end{frame}

\begin{frame}{Variance}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \item A single random variable $X$ is often characterised by its expected value (mean) $\mu=\rE(x)$, and its variance.
                \item Variance characterises the range of possible value that $X$ can take and is defined as
            \end{itemize}
            \begin{equation*}
                \var(X) = \sigma^2 = \sum_{i=1}^N \left(X_i - \mu\right)^2
            \end{equation*}
        \end{column}
