\section{Introduction}



\begin{frame}{Fitting Models to Data}
  \begin{itemize}
    \item A problem of great interest to cardiovascular scientists is to understand the dynamics of blood flow during the cardiac cycle
      \begin{itemize}
        \item Where are the cells, how fast are they moving, are they accelerating?
      \end{itemize}
    \item In a tracking experiment, you may be able to measure the positions of particles at set time points
    \item How do you calculate the velocity, acceleration, etc from the positions?
    \item You fit a \textbf{model} of the motion to the points, for example:
  \end{itemize}
  \begin{equation*}
    s=ut+\frac{1}{2}at^2, 
  \end{equation*}
  where the unknowns are $u$, the initial velocity and $a$, the acceleration. These are \textbf{free parameters} that can be chosen so that the model describes the data well.
\end{frame}

\begin{frame}{Types of Model}
    \begin{itemize}
    \item How we find the optimal values of the unknowns depends on the type of model
    \end{itemize}

  \begin{minipage}{0.5\textwidth}
    \begin{itemize}
    \item We will consider models of three different classes
      \begin{itemize}
      \item Linear models: $y=mx+c$, $s=ut+\frac{1}{2}at^2$
      \item Linear models with measurement error.
      \item Nonlinear models: $y=a+b\sin(cx)$, $y=axe^{-a^2x}$
      \end{itemize}
    \end{itemize}

\section{Least Squares Fitting}

\begin{frame}{Least Squares Fitting}
  \begin{itemize}
    \item Minimise ``difference'' between model and data
    \item Data: $\mathcal{D} = \{(x_1,y_1),\dots,(x_n,y_n)\}$
    \item Model: $y = f(\bp,x)$
    \item Difference (\textbf{residual}): $r_i(\bp) = y_i-f(\bp,x_i)$
    \item In terms of which the \textbf{least squares error}  is
      \begin{equation}
        e(\bp) = \sum_{i=1}^{n}r_i^2 = \br^\rT\br
      \end{equation}
    \item Goal: find the value of $\bp=\bp^*$ that minimises $e(\bp)$:
      \begin{equation}
        \bp^* = \argmin_\bp e(\bp)\label{eq:pstar}
      \end{equation}
  \end{itemize}

  \end{frame}

  \end{center}
\end{frame}

\section{Linear Models}
\begin{frame}{Linear Models}
  \begin{itemize}
    \item Not just straight lines!
    \item Any function linear in the unknown parameters. 
      \begin{equation}
        f(\bp,x) = p_1f_1(x)+\dots+p_mf_m(x) = \sum_{j=1}^m p_j f_j(x) \label{eq:linmodel}
      \end{equation}
    \item The residuals are therefore
      \begin{equation}
        r(\bp,x_i) = y_i - \sum_{j=1}^m p_j f_j(x_i)
      \end{equation}
    \item Treating $\bp$ as a column vector and defining matrix f $\bF$ as $F_{ij}=f_j(x_i)$ we have
      \begin{equation}
        \br = \by-\bF\bp
      \end{equation}
      where $\by=\left(y_1;\dots;y_n\right)$.
  \end{itemize}
\end{frame}

\begin{frame}{Example: Quadratic Model}
  \begin{itemize}
    \item Quadratic model $f(\bp,x)= p_1x^2+p_2x+p3$.
    \item Each row of $\bF$ corresponds to a data point
    \item Each column corresponds to a term in the model:
      \begin{eqnarray*}
        \br&=&\by-\bF\bp \\
        \begin{pmatrix}
          r_1 \\ r_2 \\ \vdots \\ r_n
        \end{pmatrix}
        & = &
        \begin{pmatrix}
          y_1 \\ y_2 \\ \vdots \\ y_n
        \end{pmatrix}
        -
        \begin{pmatrix}
          x_1^2 & x_1 & 1 \\
          x_2^2 & x_2 & 1 \\
          &\vdots &\\
          x_n^2 & x_n & 1 \\
        \end{pmatrix}
        \begin{pmatrix}
          p_1 \\ p_2 \\ p_3
        \end{pmatrix}.
      \end{eqnarray*}
      % In Matlab, we can form the matrix $F$ in this case as
      %\begin{lstlisting}
      %  F = [x.^2, x, x.^0];
      %\end{lstlisting}

    \item Our least-squares error can therefore be written
      \begin{equation}
        e(\bp) = \sum_i r_i^2=\br^\rT\br  = \left(\by-\bF\bp\right)^\rT \left(\by-\bF\bp\right)
      \end{equation}
  \end{itemize}
\end{frame}

\section{Finding the Best Fit}

\begin{frame}{Solving for the Best Fit}
  \begin{itemize}
    \item ``Best fit'' $\rightarrow$ find $\bp=\bp^*$ that minimises $e(\bp)$.
    \item Differentiate $e(\bp)$ with respect to $\bp$ and set to zero:
      \begin{equation*}
        \bF^\rT\by-\bF^\rT\bF\bp^*=0
      \end{equation*}
    \item Simultaneous equations: solve by direct inversion:
      \begin{equation*}
        \bp^* = \inv{\left(\bF^\rT\bF\right)}\bF^\rT\by
      \end{equation*}
    \item or in Matlab using Gaussian elimination through the \texttt{\textbackslash} operator:
      %\begin{lstlisting}
       % pstar = (F'*F)\(F'*y)
      %\end{lstlisting}
    \item Why not solve $\by=\bF\bp$ for $\bp$ directly?
      \begin{itemize}
      \item Guarantees on specific solution (least squares)
      \item Incorporate extra information into $e(\bp)$
      \end{itemize}
  \end{itemize}
\end{frame}

\section{Weighted Linear Least Squares}

\begin{frame}{Weighted Least Squares}
  \begin{itemize}
    \item Experimental data $\rightarrow$ multiple repeats. How do we deal with this?
    \item Fit models to the ``average''  but does the spread (variance) matter?
    \item Given less influence to ``noisy/uncertain'' data by modifying $e(\bp)$
      \begin{equation*}
        e(\bp) = \sum_{i=1}^n w_ir_i^2
      \end{equation*}
    \item Set $w_i = \frac{1}{\var(y_i)}$
    \item ``Certain'' points influence the fit more strongly than ``uncertain'' ones.
  \end{itemize}
\end{frame}

\begin{frame}{Weighted Least Squares}
  \begin{center}
    \includegraphics[width=\textwidth]{weighted}
  \end{center}
\end{frame}

\begin{frame}{Weighted Least Squares}
  \begin{itemize}
    \item Covenient to define matrix
      \begin{equation*}
        \bSig = \diag\left(\var(y_1), \var(y_1), \dots, \var(y_n) \right) 
      \end{equation*}
    \item Special case of the \textbf{covariance matrix}
    \item The error becomes
      \begin{equation*}
        e(\bp) = \br^\rT \inv{\bSig} = \left(\by-\bF\bp\right)^\rT \inv{\bSig} \left(\by-\bF\bp\right)
      \end{equation*}
      \item Note that $\inv{\bSig}$ can be done by inverting the diagonal elements 
      \item Differentiate and set to zero
        \begin{equation}
          \bF^\rT\inv{\bSig}\by-\bF^\rT\inv{\bSig}\bF\bp^*=0,
        \end{equation}
      \item Solve as before 
        \begin{equation}
          \bp^* = (\bF^\rT\inv{\bSig}\bF)^{-1}\bF^\rT\inv{\bSig}\by
        \end{equation}
  \end{itemize}
\end{frame}

\section{Fitting Nonlinear Models}
\begin{frame}{Fitting Nonlinear Models}
  \begin{itemize}
    \item What about non-linear models like $y(x) = a\sin bx$?
    \item No matrix representation (non-linear!)
    \item Cannot minimise least square error directly.
    \item Instead, focus on \textbf{reducing} the error iteratively
    \item We will turn non-linear problems into locally linear ones.
  \end{itemize}
\end{frame}

\begin{frame}{Linearising Non-linear Problems}
  Add figure in here that shows how to Taylor-expand a 1d function
  \begin{itemize}
  \item Taylor Expansion
    \begin{equation*}
      y(x+\Delta x) = y(x) + \frac{\rd y}{\rd x}\Delta x
    \end{equation*}
  \end{itemize}
\end{frame}

\begin{frame}{Linearising Non-linear Problems}
  \begin{itemize}
    \item Taylor expansion of our fitting function
      \begin{eqnarray*}
        f(x_i,\bp+\bdp) &\approx& f(x_i,\bp) + \sum_{j=1}^m \frac{\partial f(x_i)}{\partial p_j}\Delta p_j\\
        &=& f(x_i,\bp) + \sum_{j=1}^m J_{ij}\delta p_j
      \end{eqnarray*}
      \item Writing $f_i(\bp) = f(x_i,\bp)$ give $\bf(\bp+\bdp)=\bf(\bp)+\bJ\bdp$ 
      \item The residual is $\br(\bp+\bdp) = \by - \bf(\bp) -\bJ\bdp$
      \item And the error becomes
        \begin{equation*}
          e(\bp+\bdp) = \left(\by - \bf(\bp) -\bJ\bp\right)^\rT\left(\by - \bf(\bp) -\bJ\bp\right).
        \end{equation*}
      \item Minimise this $\rightarrow$ change in $\bp$ that maximally reduces the error
      \item Differentiate and set to zero:
        \begin{equation*}
          \left(\bJ^\rT\bJ\right)\bdp^* = \bJ^\rT\left(\by-\bf(\bp)\right)
        \end{equation*}
      \item Solve, repeat until finished!
  \end{itemize}
\end{frame}

\begin{frame}{Stabilising the Optimisation}
  \begin{itemize}
    \item Common modification:
      \begin{equation*}
        \left(\bJ^\rT\bJ+\lambda\diag(\bJ^\rT\bJ)\right)\bdp^* = \bJ^\rT\left(\by-\bf(\bp)\right)
      \end{equation*}
    \item Without this modification, the parameter vector will change quickly along directions with steep gradients
    \item But slowly along shallow gradients.
    \item Modification increases movement along shallow gradients and thus faster convergence.
    \item ``Damping factor'' $\lambda$ controls the extent of this.
    \item Modified iteratively to give the best results. 
    \item The full algorithm is named after its inventors: \textbf{Levenberg-Marquadt}
  \end{itemize}
\end{frame}

\begin{frame}{The Levenberg-Marquadt Algorithm}
\scalebox{0.37}{
\begin{minipage}{\linewidth}

\begin{algorithm2e}[H]
%  \caption{The Levenberg-Marquadt Algorithm}\label{alg:levmar}
%  \begin{algorithmic}[1]
  \SetLine

  \SetKwComment{com}{\% }{}
  \KwData{Data $\bx,\by$; Initial parameter guess $\bp_0$; Function $f(x,\bp)$; Jacobian $Jac(x,\bp)$; initial values of $\lambda, \nu$.}
  \KwResult{$\bp,e(\bp)$, parameters that minimmise the least squares error, the error.}
  \com{Evaluate the error at the initial parameter value}
  $\bp\gets\bp_0$\; 
  $e\gets(\by-f(x,\bp)^\rT(\by-f(x,\bp)$\;
  \com{Iterate until convergence}
  $Converged\gets False$\;
  \While{$Converged==False$}{
    \com{Calculate the Jacobian}
    $J\gets Jac(x,\bp)$\;
    \com{Calculate update for different values of $\lambda$ until we improve the error}
    $LambdaSet\gets false$\;
    \While{LambdaSet==false}{
      \com{Calculate $\Delta\bp$ for $\lambda=\lambda_0$ and $\lambda=\lambda_0/\nu$}
      $\Delta_1 =   \left(\bJ^\rT\bJ+\lambda\diag(\bJ^\rT\bJ)\right) \setminus \bJ^\rT\left(\by-\bf(\bp)\right)$\%;
      $\Delta_2 =   \left(\bJ^\rT\bJ+(\lambda/\nu)\diag(\bJ^\rT\bJ)\right) \setminus \bJ^\rT\left(\by-\bf(\bp)\right)$\;
      \com{Calculate the error at the new parameter values}
      $e_1\gets(\by-f(x,\bp+\Delta_1)^\rT(\by-f(x,\bp)$\;
      $e_2\gets(\by-f(x,\bp+\Delta_2)^\rT(\by-f(x,\bp)$\;
      \uIf{$e_1<e$}{
        \com{Set new values of $\bp$ and $e$, storing old value as $e'$}
        $\bp\gets\bp+\Delta_1$\;
        $e'\gets e$\;
        $e\gets e_1$\;
        \com{Current $\lambda$ is OK}
        $LambdaSet\gets true$\;
      }
      \uElseIf{$e_2<e$}{
        \com{Set new values of $\bp$ and $e$, storing old value as $e'$}
        $\bp\gets\bp+\Delta_2$\;
        $e'\gets e$\;
        $e\gets e_2$\;
        $\lambda\gets\lambda/\nu$\;
        \com{Current $\lambda$ is OK}
        $LambdaSet\gets true$\;
      }
      \uElse{
        $\lambda\gets\lambda/\times nu$\;
      }
    }
    \If{$e'/e<TerminationCriterion$}{
      $Converged \gets true$
    }
  }

%  \label{alg:levmar}
\end{algorithm2e}
\end{minipage}
}
\end{frame}
