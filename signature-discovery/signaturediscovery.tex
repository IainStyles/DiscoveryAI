                \begin{itemize}
                    \item Signatures can represent combinations of factors
                    \item Not always directly measurable
                    \item Not always informed by causal links
                    \item Not always selected for predictive/diagnostic ability
                    \item BUT -- often informative about underlying phenomena -- \textbf{discovery}
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}

\begin{frame}{The Basic Idea}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item Given the labelled data points:
        \item Construct a \textit{classification rule} as some combination of the input variables that separates the predefined classes as much as possible: \textbf{Discrimination}.
        \item Use the classifier to \textit{predict} the class of new unlabelled data points: \textbf{Classification}.
      \end{itemize}
    \end{column}

    \end{column}

\begin{frame}{How LDA does this}
  \begin{itemize}
    \item Construct a \textit{statistical model} of each class
    \item Each class defined by its \textit{mean} and \textit{covariance}
    \item Compute the line of equal probability
    \item Classify points according to which side of the line they are
    \pause
    \item \textbf{The statistics define the signature of the class}
  \end{itemize}
\end{frame}


\begin{frame}{Quick Review of Conditional Probability}
  We will treat this as a statistical problem, making liberal use of \textbf{Bayes' theorem}:

  \begin{equation*}
    P(a|b) = P(b|a)P(a)/P(b)
  \end{equation*}

  \begin{itemize}
    \item $P(a|b)$ is the probability of $a$ being true, given that $b$ is already true.
    \item $P(b|a)$ is the probability of $b$ being true, given that $a$ is already true.
    \item $P(a)$ and $P(b)$ are the probabilities of $a$ and $b$ being true, irrespective of prior knowledge.
  \end{itemize}
\end{frame}

\begin{frame}{Conditional Probability}
  Example: chemical ``assay'' that determines the presence of a particular chemical in a sample. We know the following::
  \begin{itemize}
    \item \textbf{Sensitivity} (true positive rate) is 90\%: if a sample contains the chemical, there is a 90\% chance of the test being positive.
    \item \textbf{Specificity} (true negative rate) is 80\%: if a sample does not contain the chemical, there is an 80\% chance of the test being negative.
    \item It is \textbf{expected (prior knowledge)} that 10\% of samples contain the chemical.
  \end{itemize}
\end{frame}

\begin{frame}{Conditional Probability}
  How does this affect test interpretation? Letting $T$ be the test and $S$ the sample we have:
  \begin{itemize}
    \item Sensitivity of 90\%
      \begin{equation*}
        \mapsto\, P(+T|+S) = 0.9\, \mbox{and}\, P(-T|+S)=0.1
      \end{equation*}
    \item Specificity of 80\%
      \begin{equation*}
        \, \mapsto\, P(-T|-S) = 0.8\, \mbox{and}\, P(+T|-S)=0.2
      \end{equation*}
    \item Prior knowledge
      \begin{equation*}
        \, \mapsto\, P(+S) = 0.1\, \mbox{and hence}\, P(-S)=0.9
      \end{equation*}
  \end{itemize}
  [sample must always give a positive or negative test result; probabilities must sum to one]
\end{frame}

\begin{frame}{Asking Questions with Conditional Probabilities}
  \textit{What is the probability that a sample contains the chemical given a positive test result?}
  \begin{itemize}
    \item From Bayes' theorem:
      \begin{equation*}
        P(+S|+T) = P(+T|+S)P(+S)/P(+T)
      \end{equation*}
    \item Need probability of positive test, $P(+T)$.
    \item Two ways to get $T$: a false positive from negative sample is negative; true positive from positive sample:
      \begin{eqnarray*}
        P(+T) &=& P(+T|+S)P(+S)+P(+T|-S)P(-S)\\  &=& (0.9\times0.1)+(0.2\times 0.9) = 0.27
      \end{eqnarray*}
      \item Similarly for a negative test:
        \begin{equation*}
          P(-T) = 0.73.
        \end{equation*}
  \end{itemize}
\end{frame}

\begin{frame}{Asking Questions with Conditional Probabilities}
  \begin{itemize}
     \item Then we have
       \begin{equation*}
         P(+S|+T) = 0.9\times0.1/0.27 = 0.333
        \end{equation*}
       and similarly
       \begin{eqnarray*}
          P(+S|-T) &=& 0.1\times0.1/0.73 = 0.014\\
          P(-S|+T) &=& 0.2\times0.9/0.27 = 0.667\\
          P(-S|-T) &=& 0.9\times0.9/0.73 = 0.986
       \end{eqnarray*}
  \end{itemize}
\end{frame}

\begin{frame}{Implications}
  \begin{itemize}
    \item If a test is negative, 98.6\% prob that the sample does not contain the chemical.
    \item If the test is positive, it is \textbf{twice as likely} that the sample will be negative than positive.
    \item Surprising given the high sensitivity of the test
    \item Must take into account the relative sizes of the two populations
    \item Only 10\% samples likely to be positive
    \item Much more likely that positives are false positives (90\% of samples ae negative)
  \end{itemize}
\end{frame}

\begin{frame}{Linear Discriminant Analysis}
  \begin{itemize}
    \item Bayesian statistics will allow us to classify data in a principled way
    \item Allows us to calculate probabilities of points belonging to groups
    \item Simplest base: linear binary classifier
    \item Linear combination of variables--a straight line--that best discriminates between two pre-defined groups of points
    \item Uses the probability distributions of the classes to find the line of equal probability dividing two classes
  \end{itemize}
\end{frame}

\begin{frame}{Formulating LDA}
  \begin{itemize}
    \item Consider a data point $\bx$ which we want to assign to one of two predefined classes $\Pi_i$ ($i=1,2$)
    \item We write that $P(\bx\in\Pi_i) = \pi_i$
    \item We also define the conditional probability $P(\bx|\Pi_i)=f_i(\bx)$, the distribution of points given class $\Pi_i$
    \item Then the probability that a given point $\bx$ belongs to class $\Pi_i$ is, by Bayes rule:
      \begin{eqnarray*}
        P(\Pi_i|\bx)  &=& \frac{P(\bx|\Pi_i)P(\Pi_i)}{P(\bx)}\\
        &=& \frac{f_i(\bx)\pi_i}{f_1(\bx)\pi_1+f_2(\bx)\pi_2}
      \end{eqnarray*}
  \end{itemize}      
\end{frame}

\begin{frame}{Formulating LDA}
  \begin{itemize}
    \item Give the distributions of the groups $f_i(\bx)$, and the probabilities $\pi_x(\bx)$ of a point being in the groups.
    \item New points are assigned to the group with the highest probability:
      \begin{equation*}
        \frac{P(\Pi_1|\bx)}{P(\Pi_2|\bx)}>1 \,\mapsto\, \bx\in\Pi_1 \quad \mbox{else}\, \bx\in\Pi_2
      \end{equation*}
    \item In terms of $f$ and $\pi$, $\bx$ belongs to $\Pi_1$ if
      \begin{equation*}
        \frac{f_1(\bx)}{f_2(\bx)}>\frac{\pi_1}{\pi_2}\label{eq:classratio}
      \end{equation*}
    \item otherwise to $\Pi_2$. 
    \item Equal ratios, randomly assign to either class.
  \end{itemize}
\end{frame}

\begin{frame}{Defining the Classes}
  \begin{itemize}
    \item We need to define the class distributions $f_i(\bx)$.
    \item For a simple two-class binary classifier we assume:
      \begin{itemize}
      \item Each data point belongs to exactly one of exactly two distinct and identifiable groups, $\Pi_1$ and $\Pi_2$
      \item The two groups are normally distributed with different means $\bar\bx_1$ and $\bar\bx_2$ but identical covariances $\mathbf{\Sigma}$.
      \end{itemize}
    \item Thus, for $n=\{1,2\}$ the groups distributions are $P(\bx|\Pi_i) = f_i(\bx)$:
      \begin{equation*}
        f_n(\bx) = \frac{1}{(2\pi)^{r/2}\abs{\mathbf{\Sigma}}^{1/2}}\exp{[-\frac{1}{2}(\bx-\bar\bx_n)^\rT\mathbf{\Sigma}^{-1}(\bx-\bar\bx_n)]}
      \end{equation*}
  \end{itemize}
\end{frame}

\begin{frame}{The Separation Rule}
  \begin{itemize}
  \item We know the ratio $\frac{f_1(\bx)\pi_2}{f_2(\bx)\pi_1}=1$ separates the groups.
  \item Taking logs: $\log_e\frac{f_1(\bx)\pi_2}{f_2(\bx)\pi_1} = \log_e\frac{f_1(\bx)}{f_2(\bx)}-\log_e\frac{\pi_1}{\pi_2}$ with:
    \begin{equation*}
      \log_e\frac{f_1(\bx)}{f_2(\bx)} = \left(\bar\bx_1-\bar\bx_2\right)^\rT\mathbf{\Sigma}^{-1}\bx -\frac{1}{2}\left(\bar\bx_1-\bar\bx_2\right)^\rT\mathbf{\Sigma}^{-1}\left(\bar\bx_1+\bar\bx_2\right)\label{eq:logp1p2}
    \end{equation*}
  \item The second term on the RHS of this equation is ``constant'' (no $\bx$) and simplifies as
    \begin{equation*}
      \left(\bar\bx_1-\bar\bx_2\right)^\rT\mathbf{\Sigma}^{-1}\left(\bar\bx_1+\bar\bx_2\right)  =\bar\bx_1^\rT\mathbf{\Sigma}^{-1}\bar\bx_1 - \bar\bx_2^\rT\mathbf{\Sigma}^{-1}\bar\bx_2
    \end{equation*}
  \end{itemize}
\end{frame}

\begin{frame}{The Separation Rule}
  \begin{itemize}
    \item Writing $\mathbf{m}=\mathbf{\Sigma}^{-1} \left(\bar\bx_1-\bar\bx_2\right)$
    \item And $c = -\frac{1}{2}\left(\bar\bx_1^\rT\mathbf{\Sigma}^{-1}\bar\bx_1 - \bar\bx_2^\rT\mathbf{\Sigma}^{-1}\bar\bx_2\right)+\log_e\frac{\pi_1}{\pi_2}$
    \item We have
      \begin{equation*}
        L(\bx) = \log_e\frac{f_1(\bx)\pi_1}{f_2(\bx)\pi_2} = \mathbf{m}^\rT\bx+c,\label{eq:lindisc}
      \end{equation*}
    \item Which yields the following decision rule:
      \begin{equation*}
        \mbox{if}\,L(\bx)>0\,\mbox{assign $\bx$ to $\Pi_1$ else $\Pi_2$}
      \end{equation*}
    \item This is \textit{Gaussian LDA}
    \item $\mathbf{m}^\rT\bx$ is \textit{Fisher's linear discriminant function}.
  \end{itemize}
\end{frame}

\begin{frame}{Assumptions of LDA}
  \begin{enumerate}
    \item There are two classes
    \item There are sufficient labelled examples to estimate the class statistics
    \item Classes are cleanly separated by a linear boundary
    \item Both classes are normally distributed 
    \item Both classes having the same covariance
  \end{enumerate}
  \pause
  There are extensions of LDA to address assumptions 1, 3 and 5

  In practice, some relaxation of conditions 4, 5 is possible without too much impact.
\end{frame}

\begin{frame}{Quadratic Discriminant Analysis}
  \begin{itemize}
    \item Different class covariances require more complex analysis
    \item If the covariances of the two classes are $\mathbf{\Sigma}_1$ and $\mathbf{\Sigma}_2$ the discriminant becomes:
      \begin{equation*}
        Q(\bx) = \bx^\rT\bA\bx + \bb^\rT\bx + c
      \end{equation*}
      a \textit{Quadratic discriminant} with
      \begin{eqnarray*}
        \bA &=& -\frac{1}{2}\left(\mathbf{\Sigma}_1^{-1}-\mathbf{\Sigma}_2^{-1}\right) \\
        \bb &=& \mathbf{\Sigma}^{-1}_1\bar\bx_1 - \mathbf{\Sigma}^{-1}_2\bar\bx_2 \\
        c &=& -\frac{1}{2}\left(\log_e\frac{\abs{\mathbf{\Sigma}_1}}{\abs{\mathbf{\Sigma}_2}}+\bar\bx_1^\rT\mathbf{\Sigma}_1^{-1}\bar\bx_1-\bar\bx_2^\rT\mathbf{\Sigma}_2^{-1}\bar\bx_2\right)-\log_e\frac{\pi_1}{\pi_2}
      \end{eqnarray*}

      The classification rule is:
      \begin{equation*}
        \mbox{if}\,Q(\bx)>0\,\mbox{assign $\bx$ to $\Pi_1$}
      \end{equation*}
      else assign $\bx$ to $\Pi_2$. 
  \end{itemize}
\end{frame}

\begin{frame}{Multiclass LDA}
  \begin{itemize}
    \item Generalisation multiclass data require significantly more algebra\dots.
    \item Compute the pairwise relative probabilities as before and form the discriminant
      \begin{equation*}
        L_{ij}(\bx) = \log_e\left(\frac{P(\Pi_i|\bx)}{P(\Pi_j|\bx}\right)=\log_e\left(\frac{f_i(\bx)\pi_i}{f_j(\bx)\pi_j}\right)
      \end{equation*}
    \item $\bx$ is assigned to $\Pi_i$ if $L_{IJ}>0$ for all $j\neq i$. 
    \item The discriminant function between classes $i$ and $j$ is then
      \begin{equation*}
        L_{ij}(\bx) =\mathbf{m}^\rT_{ij}\bx+c_{ij}
      \end{equation*}
      with 
      \begin{eqnarray*}
        \mathbf{m}_{ij} &=& \left(\bar\bx_i-\bar\bx_j\right)^\rT\mathbf{\Sigma}^{-1}\, \mbox{and}\\
        c_{ij}&=&-\frac{1}{2}\left(\bar\bx_i^\rT\mathbf{\Sigma}^{-1}\bar\bx_i - \bar\bx_j^\rT\mathbf{\Sigma}^{-1}\bar\bx_j\right)+\log_e\frac{\pi_i}{\pi_j}.
      \end{eqnarray*}
  \end{itemize}
\end{frame}

\begin{frame}{The $k$-means algorithm}
  \begin{itemize}
    \item In $k$-means clustering we aim to partition set of $n$ observations into $k \leq n$ sets $\mathbf{S} = \{S1, S2, ..., S_k\}$
    \item This is done to minimize the within-cluster sum of squares (the variance)
    \item In other words, we try to find the $k$ ``tightest'' clusters
    \item Formally this requires us to find:
    \begin{equation*}
      \argmin_\mathbf{S} \sum_{i=1}^k\sum_{\mathbf{x\in S_i}}\left\|\mathbf{x} -\bm{\mu}_{i}\right\|^{2}
    \end{equation*}
    \pause
    \item This is a surprisingly hard problem to solve in general
    \begin{itemize}
      \item Combinatorial optimisation does not scale well
    \end{itemize}
    \item In practice we solve using a heuristic
  \end{itemize}
\end{frame}

\begin{frame}{The $k$-means Algorithm}
  \begin{columns}[t]
    \begin{column}{0.48\textwidth}
      \begin{block}{Informally\dots}
        \begin{itemize}
          \item Number of clusters $k$ is specified by the ``user''. 
          \begin{itemize}
            \item[1)] First, randomly assign every data point to one of the clusters.
            \item[2)] Compute the mean position (in the data space) of each cluster
            \item[3)] For each data point, calculate distance to mean of each cluster
            \item[4)] Re-assign points to their nearest cluster.
            \item[5)] Return to step 2 and repeat until the clusters are not changing.
          \end{itemize}
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Lloyd's Algorithm}
        \scalebox{0.2}{
          \begin{minipage}{\linewidth}
            \begin{algorithm}[H]
              \SetKwComment{com}{\% }{}
              \KwData{Data points $Data[N]$; Number of clusters $K$}
              \KwResult{$\bp,e(\bp)$, parameters that minimmise the least squares error, the error.}

              \com{Randomly assign each point to a cluster}
              \For{$i\gets 1$ \KwTo $N$}{
                $ClusterIndex[i]\gets randint(1,K)$
              }
              \com{Calculate the mean of each cluster}
              \For{$i\gets 1$ \KwTo $K$}{
                $ClusterMean[i] \gets mean(Data[Cluster[i]])$
              }

              \com{Iterate until clusters are stable}
              $Converged\gets False$\;
              \While{$Converged==False$}{

                  \com{Compute distance from every point to each cluster mean}
                  \For{$i\gets 1$ \KwTo $K$}{
                    \For{$j\gets 1$ \KwTo $N$}{
                      $dist[i][j] \gets distance(ClusterMean[i],Data[j])$
                    }
                  }

                  \com{Find the nearest mean for each data point}
                  \For{$i\gets 1$ \KwTo $N$}{
                    \For{$j\gets 1$ \KwTo $K$}{
                      $NewClusterIndex[i] = nearest(Data[i],ClusterMean[j])$
                    }
                  }

                  \com{Generate new clusters}
                  \For{$j\gets 1$ \KwTo $K$}{
                    $ClusterMean[i] \gets mean(Data[NewClusterIndex==i])$
                  }

                  \com{Check to see if cluster membership has changed}
                  $Converged\gets True$

                  \If{NewClusterIndex != ClusterIndex}{
                      $Converged \gets False$
                  }

                  \com{Replace old clusters with new ones}
                  $ClusterIndex \gets NewClusterIndex$
                  $ClusterMean \gets NewClusterMean$
                }
              \caption{The $k$-Means Algorithm.}
              \label{alg:kmeans}
            \end{algorithm}
          \end{minipage}
        }
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Notes about $k$-means}
  There are a few points to note:
  \begin{itemize}
    \item Initial random seeding $\rightarrow$ repeats needed
    \item Have to ``guess'' the number of clusters.
    \begin{itemize}
      \item Too many, and you will split what should be a single cluster.
      \item Too few, and you will combine clusters that should be separate, or splitting clusters and merge with others.
    \end{itemize}
    \item This is a computationally ``hard'' algorithm (in the jargon, it is NP-hard).
      \begin{itemize}
        \item The number of distance calculations increases rapidly with the number of clusters and the number of data points so it can be very costly for large datasets.
        \item But not as hard as methods that compare all data points with each other
      \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Principal Component Analysis}

\begin{frame}{Principal Component Analysis}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item PCA allows us to find a decomposition that aligns components to the shape of the data:
        \begin{itemize}
          \item The first coordinate is aligned with the direction of greatest variation in the data.
          \item The second coordinate is perpendicular to the first coordinate and aligned with the direction of greatest variation under that constraint,
          \item Subsequent coordinates are perpendicular to all that have already been defined, aligned with the direction of next-greatest variation under that constraint.
        \end{itemize}
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \includegraphics[width=\textwidth]{pcasketch}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Performing PCA}
  \begin{itemize}
    \item Covariance matrix, $\mathbf{\Sigma}$ is central to PCA
      \begin{equation*}
        \Sigma_{ij} = \frac{1}{N-1}\sum_{n=1}^N \left(x_i^{(n)}-\mu_i\right)\left(x_j^{(n)}-\mu_j\right)
      \end{equation*}
    \item Easy to compute $\mathbf{\Sigma}$ if we organise our dataset in a sensible way:
      \begin{itemize}
        \item Construct matrix $\bX$ with components $X_{ij} = x_i^{(j)}$
        \item Each \textit{row} of $\bX$ corresponds to a variable
        \item Each \textit{column} corresponds to a sample
        \item Compute the mean of every row and tile the resulting column vector side-by-side $N$ times $\rightarrow\bar\bX$.
        \item Then, we form the covariance matrix as
          \begin{equation}
            \mathbf{\Sigma} = \frac{1}{N-1}\left(\bX-\bar\bX\right)\left(\bX-\bar\bX\right)^\rT = \frac{1}{N-1}\tilde\bX\tilde\bX^\rT
          \end{equation}
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Performing PCA}
  \begin{itemize}
    \item We will now find coordinate vectors that align with the covariance structure
    \item Define matrix $\bT=\begin{pmatrix}\bt_1 & \bt_2 & \dots \bt_M\end{pmatrix}$
    \item Columns of $\bT$ (the vectors $\bt$) represent  form a new coordinate system (basis) for the space.
    \item Normalised such that $\bT^\rT\bT=\bI\leftrightarrow \bt_i\cdot\bt_j = \delta_{ij}$.
    \item First we find the direction of greatest variance
    \item Find the vector $\bt_1$ that maximises the variance of the projection of the data.
    \begin{equation*}
      \bt_1 =  \argmax_{\bt}\var(\bt^\rT\bX)
    \end{equation*}
    \item Now $\var(\bt^\rT\bX)=\bt^\rT\tilde\bX\tilde\bX^\rT\bt = \bt^\rT\mathbf{\Sigma}\bt$ ignoring the pre-factor $1/(N-1)$
    \item Find $\bt=\bt_1$ that maximises this subject to $\bt^\rT\bt=1$
  \end{itemize}
\end{frame}

\begin{frame}{Performing PCA}
  \begin{itemize}
    \item Mathematical ``trick'': Lagrange multiplier - allows us to enforce constraints
    \item Subtract a ``multiple of zero'':
      \begin{equation}
        \bt_1 = \argmax _\bt \left[\bt^\rT\mathbf{\Sigma}\bt -\lambda_1(\bt^\rT\bt-1)\right]
      \end{equation}
    \item Differentiate this with respect to $\lambda$ and set to zero to maximise
      \begin{equation}
        \left(\mathbf{\Sigma}-\lambda_1\bI\right)\bt_1 = 0\quad\rightarrow\quad \mathbf{\Sigma}\bt_1 = \lambda_1\bt_1
      \end{equation}
    \item Solve ''\textit{eigenvalue}'' problem for largest $\lambda_1$ and accompanying $\bt_1$
    \item $\bt_1$ is direction of greatest variation
    \item $\bt_1^\rT\mathbf{\Sigma}\bt_1=\lambda_1\bt_1^\rT\bt_1 = \lambda_1$ is variance along that direction 
    \item Next-greatest direction of variation is given by the second-largest eigenvalue/vector pair\dots
  \end{itemize}
  \begin{quote}
    \textbf{The principal components of a dataset are the eigenvectors of the covariance metric, ordered by eigenvalue.}
  \end{quote}
\end{frame}

\begin{frame}{}
  \begin{itemize}
    \item Once PCs have been found we transform the zero-mean data into the new coordinate system
    \begin{equation}
      \bX' = \bT^\rT\tilde\bX
    \end{equation}
    \item $\sqrt(\lambda_i)\bt_i$ are called the \textbf{loadings}. 
    \item Columns of $\bX'$ are the \textbf{scores}: data projected onto the PC axes.
    \item Note that $\mathbf{\Sigma} = \bT\bS\bT^\rT$ where $\bS=\diag(\lambda_1,\lambda_2,\dots,\lambda_n)$.
  \end{itemize}
\end{frame}

\begin{frame}{Reducing dimensionality with PCA}
  \begin{itemize}
    \item Each PC is a linear combination of the original variables
    \item Because PC's are aligned with data, do not always need all of them
    \item We can remove non-varying dimensions.
    \item Useful to consider the ``total variance'' $V_\rT = \sum_i \lambda_i$.
    \item Compute proportion of total variance along each PC by computing $\lambda_i/V_\rT$. 
    \item Determine how many PCs describe the majority of the variance. 
    \item Decide which ones to remove
    \item Set components of data in those dimensions to zero or equivalently, remove the corresponding columns from $\bT$.
  \end{itemize}
\end{frame}

\begin{frame}{Data Normalisation and Variable Scaling}
  \begin{itemize}
    \item In some multivariate datasets, variables are of a different type or on a different scale
    \item Useful to consider mapping all variables onto a common scale to avoid bias
    \item Most common: rescale all variables to have unit variance.
    \item Take care when using PCA in such a situation.
  \end{itemize}
\end{frame}

\begin{frame}{PCA and signatures}
  \begin{itemize}
    \item The components (loadings) of PCA can be regarded as signatures
    \item Each component represents some fundamental building block of the data
  \end{itemize}
\end{frame}

\begin{frame}{Non-Negative Matrix Factorisation}

    \begin{column}{0.78\textwidth}
      \begin{itemize}
        \item NNMF is not as clean as PCA
        \item No analytic solutions
        \item Requires approximation and numerics
        \item NNMF is typically an \textbf{approximate} decomposition
        \only<1>{

          \begin{equation*}
            X^{(i)}_j \approx \sum_{k=1}^K A_{ik}\Phi^{(k)}_j
          \end{equation*}
          with $A_{ik}$ and $\Phi^{(k)}_j$ all strictly $\geq 0$.
        }

      \end{itemize}

\begin{frame}{Computing analysis NNMF}
  \begin{itemize}
    \item The general problem statement is $\mathbf{X}\approx \mathbf{A}\bm{\Phi}$ with $\bX$ known
    \item The elements of $\mathbf{X}$, $\mathbf{A}$, $\bm{\Phi}$ are all $\ge 0$
    \item Need to solve $\argmin_{\bA,\bm{\Phi}} \|\bX-\bA\bm\Phi\|^2$ under positivity constraints
    \item Can be solved via linear programming, convex optimisation, and other methods
  \end{itemize}
\end{frame}



\begin{frame}{Non-linear Representations}
  \begin{itemize}
    \item PCA and NNMF are purely linear transformations of the data
    \item Do not change the ``shape'' of a cloud of datapoints
    \item Not suited for finding finer structure in a dataset
    \item Popular alternatives for this are:
    \begin{itemize}
      \item Self-organising maps
      \item Curvilinear component analysis
      \item Latent variable models
      \item t-distributed stochastic neighbour embedding
      \item Uniform Manifold Approaximation and Projection (UMAP)
    \end{itemize}
    \item Often use linear methods as pre-processing to eliminate globally redundant dimensions 
    \item Then use a non-linear method or clustering to sort out the subleties.
  \end{itemize}
\end{frame}
